二、 用appid、签名初始化sdk

注：推荐在Application的onCreate方法中初始化

// 设置日志level
mZegoAVKit.setLogLevel(context, ZegoAVKit.LOG_LEVEL_DEBUG, null);

// 即构分配的app key与app id
byte[] signKey = {};
int appID = 1;

// 初始化sdk
mZegoAVKit.init(appID, signKey, context);

// 初始化配置
mZegoAVConfig = new ZegoAvConfig();
mZegoAVConfig.setResolution(640, 480);
mZegoAVKit.setAVConfig(mZegoAVConfig);

2. 登录成功后发布直播

// 发布直播

3. 停止发布直播

// 停止publish前，必须停止预览
mZegoAVKit.stopPreview();
mZegoAVKit.stopPublish();

2. 登录成功后播放视频

// 播放视频

3. 停止播放视频

mZegoAVKit.stopPlayStream(streamID);

五、退出页面时，停止发布或者停止播放

// 停止发布
if(mIsPublishing){
    mZegoAVKit.stopPreview();
    mZegoAVKit.stopPublish();
}

// 停止播放
if(mIsPlaying){
    mZegoAVKit.stopPlayStream(streamID);
}

// 退出频道
mZegoAVKit.logoutChannel();
