1. 支持iOS8.0以上(SDK为动态库)

2. 支持iOS7.0以上(SDK为静态库)

二、初始化SDK

// 设置日志级别, generic:3  debug:4
[ZegoLiveApi setLogLevel:4];
// 设置是否开启“测试环境”, true:开启 false:关闭
[ZegoLiveApi setUseTestEnv:false];

// Demo 把signKey先写到代码中
// 规范用法：这个signKey需要从server下发到App，避免在App中存储，防止盗用
Byte signkey[] = {0x91,0x93,0xcc,0x66,0x2a,0x1c,0xe,0xc1,
     0x35,0xec,0x71,0xfb,0x7,0x19,0x4b,0x38,
     0x15,0xf1,0x43,0xf5,0x7c,0xd2,0xb5,0x9a,
     0xe3,0xdd,0xdb,0xe0,0xf1,0x74,0x36,0xd};
NSData * appSign = [[NSData alloc] initWithBytes:signkey length:32];
uint appID = 1;

// 初始化sdk
NSData *sign = [[NSData alloc]initWithBytes:signkey length:32];
g_zegoAV = [[ZegoLiveApi alloc] initWithAppID:appID appSignature:sign];

// 设置是否开启“硬件加速”, true: 开启  false:关闭
[g_zegoAV requireHardwareAccelerated:false];

[g_zegoAV setDelegate:[ZegoLiveApiDelegateImpl alloc]]

/// @直播回调
@interface ZegoLiveApiDelegateImpl () <ZegoLiveApiDelegate>
@end
@implementation ZegoLiveApiDelegateImpl
/// \brief 登录频道的回调
- (void)onLoginChannel:(NSString *)channel error:(uint32)err {
    if（err == 0）{
        // 登录成功，开始发布直播
        [g_zegoAV setLocalView:v];
        [g_zegoAV setLocalViewMode:ZegoVideoViewModeScaleAspectFill];
        [g_zegoAV startPreview];
        [g_zegoAV startPublishingWithTitle:liveTitle streamID:streamID];
    }else{
        // 登录频道失败，提示用户
    }
}

/// \brief 发布直播成功
- (void)onPublishSucc:(NSString *)streamID channel:(NSString *)channel playUrl:(NSString *)playUrl {
    NSLog(@"%s, stream: %@, url: %@", __func__, streamID, playUrl);

九、常见问题

1. 屏幕旋转处理方式

//屏幕旋转时调用setDisplayRotation接口
- (void)setRotateFromInterfaceOrientation:(UIInterfaceOrientation)fromInterfaceOrientation
{
    switch (fromInterfaceOrientation) {
        case UIInterfaceOrientationPortrait:
            [getZegoAV_ShareInstance() setDisplayRotation:CAPTURE_ROTATE_0];
            break;

        case UIInterfaceOrientationPortraitUpsideDown:
            [getZegoAV_ShareInstance() setDisplayRotation:CAPTURE_ROTATE_180];
            break;

        case UIInterfaceOrientationLandscapeLeft:
            [getZegoAV_ShareInstance() setDisplayRotation:CAPTURE_ROTATE_270];
            break;

        case UIInterfaceOrientationLandscapeRight:
            [getZegoAV_ShareInstance() setDisplayRotation:CAPTURE_ROTATE_90];
            break;

        default:
            break;
    }
}

//ios8以上重载此方法
- (void)viewWillTransitionToSize:(CGSize)size withTransitionCoordinator:(id<UIViewControllerTransitionCoordinator>)coordinator
{
//    NSLog(@"viewWillTransitionToSize");
    [coordinator animateAlongsideTransition:^(id<UIViewControllerTransitionCoordinatorContext>  _Nonnull context) {
        UIInterfaceOrientation orientation = [[UIApplication sharedApplication] statusBarOrientation];
        [self setRotateFromInterfaceOrientation:orientation];
    } completion:^(id<UIViewControllerTransitionCoordinatorContext>  _Nonnull context) {

    }];

    [super viewWillTransitionToSize:size withTransitionCoordinator:coordinator];
}

//ios 7
- (void)willAnimateRotationToInterfaceOrientation:(UIInterfaceOrientation)toInterfaceOrientation duration:(NSTimeInterval)duration
{
//    UIInterfaceOrientation orientation = [[UIApplication sharedApplication] statusBarOrientation];
    [self setRotateFromInterfaceOrientation:toInterfaceOrientation];

    [super willAnimateRotationToInterfaceOrientation:toInterfaceOrientation duration:duration];
}

2. 来电响应处理方式

 //1.电话状态监听（监听广播AVAudioSessionInterruptionNotification）

[[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(audioSessionWasInterrupted:) name:AVAudioSessionInterruptionNotification object:nil];

//2.observer中实现audio session的开始和结束
- (void)audioSessionWasInterrupted:(NSNotification *)notification
{
    if ([notification.userInfo count] == 0)
    {
        return;
    }

    if (AVAudioSessionInterruptionTypeBegan == [notification.userInfo[AVAudioSessionInterruptionTypeKey] intValue])
    {
        if(self.isPublishing && self.liveType == 2){
            _publishAfterPlay = YES;
        }

        // 停止直播(主播方)
        if (self.isPublishing) {
            self.isPublishing = NO;
            [getZegoAV_ShareInstance() stopPreview];
            [getZegoAV_ShareInstance() stopPublishing];
        }

        // 停止播放（观众方）
        if([_playStreamArray count] > 0){
            for(int i = 0; i < [_playStreamArray count]; i++){
                [getZegoAV_ShareInstance() stopPlayStream:[_playStreamArray objectAtIndex:i]];
            }
        }

        // 退出房间
        [getZegoAV_ShareInstance() logoutChannel];

    }
    else if(AVAudioSessionInterruptionTypeEnded == [notification.userInfo[AVAudioSessionInterruptionTypeKey] intValue])
    {
        // 恢复发布或观看
    }
}

十、错误码说明

在Publish和Play的回调中出现以下错误码 说明：

错误码 说明
AVStateEnd = 1 直播正常停止
TempBroken = 2 直播临时中断
FatalError = 3 直播遇到严重的问题
CreateStreamError=4 创建直播流失败
FetchStreamError=5 获取流信息失败
NoStreamError=6 无流信息
MediaServerNetWorkError=7 媒体服务器连接失败
DNSResolveError=8 DNS 解释失败
